#SimuProc 1.4.3.0
      CLA
      MOV FDA,AX
      INC AX
      MOV FDB,AX
      INC AX
      MOV FDC,AX

      MSG "0 - PARA INGRESAR VECTORES"
      MSG "1 - PARA SUMAR VECTORES"
      MSG "2 - PARA RESTAR VECTORES"
      MSG "3 - PARA SALIR"
      LDT

      CMP FDC
      JMA 49; 45 SALTA SI ES 3
      CMP FDB
      JMA 35; 34 SALTA SI ES 2
      CMP FDA
      JMA 21; SALTA SI ES 1

      LDT "ENTRE TAMAÑO DEL VECTOR"
      MOV DDD,AX
      MOV CX,DDD ; LIMITE DE ELEMENTOS DEL VECTOR
      MOV BX,FDA ; BX = 0

      LDT "ENTRE UN ELEMENTO A EL VECTOR 1: "
      STB EDD
      INC BX
      LOOP 15; CICLO PARA LLENAR EL VECTOR 1

      MOV CX,DDD  ; SETEA EL LIMITE DE ELEMENTOS A LEER
      MOV BX,FDA  ; SET LA VARIABLE INCREMENTO EN 0

      LDT "ENTRE UN ELEMENTO A EL VECTOR 2: "
      STB FDD
      INC BX
      LOOP 1B; CICLO PARA LLENAR EL VECTOR 2

      MOV CX,FDC
      LOOP 0 ; LEYO LOS VECTORES AHORA VUELVE AL MENU

      MOV CX,DDD  ; CX = CANTIDAD ELEMENTOS EN EL VECTOR
      MOV BX,FDA  ; SET LA VARIABLE INCREMENTO EN 0

      MOV AX,DDD
      CMP FDB
      JME 31; SI NO SE HAN INGRESADO VECTORES

      MSG "LOS VALORES DEL VECTOR RESULTANTE SON"

      LDB EDD
      MOV CDD,AX
      LDB FDD
      ADD CDD
      EAP
      INC BX
      LOOP 27; LOOP PARA LA SUMA DE VECTORES

      INC CX
      INC CX
      LOOP 0

      MSG "NO SE HAN INGRESADO VECTORES IMPOSIBLE SUMAR"
      INC CX
      INC CX
      LOOP 0 ; SACA MENSAJE DE ERROR Y VUELVE AL MENU

      MOV CX,DDD  ; CX = CANTIDAD ELEMENTOS EN EL VECTOR
      MOV BX,FDA  ; SET LA VARIABLE INCREMENTO EN 0

      MOV AX,DDD
      CMP FDB
      JME 45; SI NO SE HAN INGRESADO VECTORES

      MSG "LOS VALORES DEL VECTOR RESULTANTE SON"

      LDB EDD
      MOV CDD,AX
      LDB FDD
      SUB CDD
      EAP
      INC BX
      LOOP 3B; LOOP PARA LA RESTA DE VECTORES

      INC CX
      INC CX
      LOOP 0

      MSG "NO SE HAN INGRESADO VECTORES IMPOSIBLE RESTAR"
      INC CX
      INC CX
      LOOP 0 ; SACA MENSAJE DE ERROR Y VUELVE AL MENU

      HLT
